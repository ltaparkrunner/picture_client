2. Обновленный сервер server.js
Теперь сервер принимает картинку через Protobuf, отправляет её в MinIO, а в MongoDB сохраняет только метаданные и путь.
javascript
const http = require('http');
const { Server } = require('socket.io');
const mongoose = require('mongoose');
const protobuf = require('protobufjs');
const { S3Client, PutObjectCommand } = require("@aws-sdk/client-s3");

// Настройки из окружения
const { S3_ENDPOINT, S3_ACCESS_KEY, S3_SECRET_KEY, S3_BUCKET, MONGO_URL } = process.env;

// 1. Подключение к MongoDB
mongoose.connect(MONGO_URL);
const ImageRecord = mongoose.model('ImageRecord', {
  name: String,
  s3Key: String,
  contentType: String,
  createdAt: { type: Date, default: Date.now }
});

// 2. Настройка S3 (MinIO) клиента
const s3Client = new S3Client({
  endpoint: S3_ENDPOINT,
  region: "us-east-1", // Для MinIO любое значение
  credentials: { accessKeyId: S3_ACCESS_KEY, secretAccessKey: S3_SECRET_KEY },
  forcePathStyle: true // Обязательно для MinIO
});

const server = http.createServer();
const io = new Server(server);

// 3. Загрузка Protobuf и запуск логики
protobuf.load("image.proto", (err, root) => {
  if (err) throw err;
  const Picture = root.lookupType("Picture");

  io.on('connection', (socket) => {
    socket.on('upload_image', async (buffer) => {
      try {
        // Десериализация
        const msg = Picture.decode(new Uint8Array(buffer));
        const { fileName, data, contentType } = Picture.toObject(msg);

        const s3Key = `${Date.now()}-${fileName}`;

        // А) Загрузка в MinIO
        await s3Client.send(new PutObjectCommand({
          Bucket: S3_BUCKET,
          Key: s3Key,
          Body: data,
          ContentType: contentType
        }));

        // Б) Запись в MongoDB
        const record = new ImageRecord({ name: fileName, s3Key, contentType });
        await record.save();

        socket.emit('status', { success: true, s3Key });
        console.log(`Uploaded to MinIO & Saved to DB: ${fileName}`);
      } catch (e) {
        console.error(e);
        socket.emit('status', { success: false, error: e.message });
      }
    });
  });
});

server.listen(3000, () => console.log('Server running on :3000'));

Для интеграции Protobuf в проект на CMake лучше всего использовать встроенный модуль FindProtobuf. Это позволит автоматически найти установленную библиотеку и сгенерировать файлы .pb.h / .pb.cc прямо во время сборки.
1. Настройка CMakeLists.txt
Добавьте эти строки в ваш файл:
cmake
cmake_minimum_required(VERSION 3.10)
project(QtImageClient)

# 1. Ищем Qt
find_package(Qt6 REQUIRED COMPONENTS Core Network WebSockets) # Или Qt5

# 2. Ищем Protobuf
find_package(Protobuf REQUIRED)

# 3. Указываем путь к .proto файлу и генерируем исходники
# Эта команда создаст переменные PROTO_SRCS и PROTO_HDRS
protobuf_generate_cpp(PROTO_SRCS PROTO_HDRS image.proto)

# 4. Создаем исполняемый файл
add_executable(QtImageClient
    main.cpp
    client.cpp
    client.h
    ${PROTO_SRCS} # Сгенерированные .cc файлы
    ${PROTO_HDRS} # Сгенерированные .h файлы
)

# 5. Подключаем пути к заголовочным файлам (включая папку со сгенерированными файлами)
target_include_directories(QtImageClient PRIVATE
    ${CMAKE_CURRENT_BINARY_DIR}
    ${Protobuf_INCLUDE_DIRS}
)

# 6. Линкуем библиотеки
target_link_libraries(QtImageClient PRIVATE
    Qt6::Core
    Qt6::Network
    Qt6::WebSockets
    ${Protobuf_LIBRARIES}
)
Используйте код с осторожностью.
2. Как это работает под капотом
find_package(Protobuf REQUIRED): Ищет установленный в системе protoc и библиотеки.
protobuf_generate_cpp: Добавляет в процесс сборки вызов protoc. Файлы будут созданы в папке сборки (build), чтобы не «мусорить» в папке с исходным кодом.
${CMAKE_CURRENT_BINARY_DIR}: Мы добавляем эту папку в include_directories, чтобы ваш код мог сделать #include "image.pb.h", не зная точного пути.
3. Совет по компиляции
Если вы используете vcpkg, не забудьте передать путь к нему при первой конфигурации проекта:
bash
cmake -B build -S . -DCMAKE_TOOLCHAIN_FILE=[путь_к_vcpkg]/scripts/buildsystems/vcpkg.cmake
cmake --build build
Используйте код с осторожностью.
4. Важный нюанс для Windows (DLL)
Если вы собираете проект на Windows и Protobuf установлен как динамическая библиотека, добавьте это перед поиском пакета, чтобы избежать ошибок линковки:
cmake
add_definitions(-DPROTOBUF_USE_DLLS)
